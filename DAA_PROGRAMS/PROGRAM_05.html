MIN AND MAX

#include <stdio.h>
int max, min;
int a[50];
void maxmin(int i, int j)
{
    int max1, min1, mid;
    if (i == j)
    {
        max = min = a[i];
    }
    else
    {
        if (i == j - 1)
        {
            if (a[i] < a[j])
            {
                max = a[j];
                min = a[i];
            }
            else
            {
                max = a[i];
                min = a[j];
            }
        }
        else
        {
            mid = (i + j) / 2;
            maxmin(i, mid);
            max1 = max;
            min1 = min;
            maxmin(mid + 1, j);
            if (max < max1)
            {
                max = max1;
            }
            if (min > min1)
            {
                min = min1;
            }
        }
    }
}
int main()
{
    int i, num;
    clrscr();
    printf("ENTER THE TOTAL NUMBER OF ELEMENTS : ");
    scanf("%d", &num);
    printf("ENTER THE NUMBERS : \n");
    for (i = 1; i <= num; i++)
    {
        scanf("%d", &a[i]);
    }
    max = a[0];
    min = a[0];
    maxmin(1, num);
    printf("MINIMUM ELEMENT IN AN ARRAY : %d \n", min);
    printf("MAXIMUM ELEMENT IN AN ARRAY : %d \n", max);
    getch();
    return 0;
}
