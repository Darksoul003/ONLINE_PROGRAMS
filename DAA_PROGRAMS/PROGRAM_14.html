#include<stdio.h>
#include<conio.h>

int add(int freq[],int i,int j)
{
    int k,s=0;
    for(k = i;k<=j;k++)
    {
        s+=freq[k];
    }
    return s;
}

int opticost(int freq[],int i,int j)
{
    int freqsum,min=999,r,opcost;
    if(j<i)
    {
        return freq[i];
    }
    freqsum = add(freq,i,j);
    for(r=i;r<=j;++r)
    {
        opcost = opticost(freq,i,r-1)+opticost(freq,r+1,j);
        if(opcost<min)
        {
            min = opcost;
        }
        return min+freqsum;
    }
    return 0;
}
