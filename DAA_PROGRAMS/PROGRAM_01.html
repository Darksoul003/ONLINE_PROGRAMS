#include<stdio.h>
#include<conio.h>
main()
{
      int n,j,temp,pos,i,a[10];
      clrscr();
      printf("ENTER THE LIMIT : ");
      scanf("%d",&n);
      printf("ENTER THE ARRAY ELEMNET : \n");
      for(i=0;i<n;i++)
      {
            scanf("%d",&a[i]);
      }
      for(i=0;i<n;i++)
      {
            pos=i;
            for(j=i+1;j<n;j++)
            {
                  if(a[j]<a[pos])
                  {
                       pos = j;
                  }
            }
            temp = a[i];
            a[i] = a[pos];
            a[pos] = temp;
      }
      printf("SORTED ARRAY IS \n",a[i]);
      for(i=0;i<n;i++)
      {
            printf("%d \n",a[i]);
      }
      getch();
      return 0;
}
